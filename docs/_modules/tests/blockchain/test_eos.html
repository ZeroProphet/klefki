[docs]def test_pub():
    for (_priv, _pub) in zip([PRIV, PRIV_1, PRIV_2], [PUB, PUB_1, PUB_2]):
        decoded = decode_privkey(_priv)
        pub = gen_pub_key(decoded)
        raw_pub = base58.b58decode(_pub[3:]).hex()
        raw_PUB = base58.b58decode(_pub[3:]).hex()
        assert raw_PUB == raw_pub
        assert _pub == pub

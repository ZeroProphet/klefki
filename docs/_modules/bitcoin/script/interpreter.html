[docs]def op_pushdata(stack, script, n):
    next_n_bytes = script[0:n]
    count = int(next_n_bytes.hex(), 16)
    stack.push(script[n: count + n])
    stack.output = next_n_bytes
    return stack.eval(script[count + n], stack, script[count + n:])
